Module Imandra_protocol.Response

type t =
| R_response of {
stamp : Stamp.t;
response : response;
}
| R_error of {
loc : Imandra_util.Iloc.t;
msg : string;
kind : Imandra_util.Error.kind;
process : string;
}

report error

| R_unsupported of Imandra_util.Iloc.t * string

report unsupported

and response =
| Res_nop

do nothing

| Res_ok
| Res_validate of Validate.Response.t
| Res_simplify of Simplify.Response.t
| Res_verify of Verify.Response.t
| Res_decompose of Decompose.Response.t
| Res_modular_decomp of Modular_decomp.Response.t
| Res_code of Imandra_surface.Code_fragment.t
| Res_ext_solvers of string list
| Res_profile of {
name : string;
probes : Imandra_util.Profile.probe list;
}
| Res_custom of {
res : string;
}
val force_lazy : t -> unit
